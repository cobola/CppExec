```cpp
// 输入：命令行参数
// 输出：标准输出(stdout)返回结果，标准错误(stderr)返回错误信息
// 返回码：0=成功，非0=失败

// 示例代码结构
int main(int argc, char* argv[]) {
    // 1. 参数校验
    if (argc != 3) {
        std::cerr << "参数错误：需要2个参数" << std::endl;

    }
    
    // 2. 业务逻辑处理

    double b = std::stod(argv[2]);
    double result = a + b;
    
    // 3. 输出结果
    std::cout << result << std::endl;
    return 0;
}
```

### 3.2 FastAPI接口层设计

#### 3.2.1 接口定义

```python
from fastapi import FastAPI, Request, HTTPException
from pydantic import BaseModel
import subprocess
import time
from typing import Optional

app = FastAPI(title="C++程序接口服务", version="1.0")

# Judge0兼容请求模型
class ExecuteRequest(BaseModel):
    source_code: Optional[str] = None  # 兼容字段
    stdin: Optional[str] = None        # 输入参数
    language_id: Optional[int] = None  # 兼容字段

# Judge0兼容响应模型
class Status(BaseModel):
    id: int
    description: str

class ExecuteResponse(BaseModel):
    code: int
    msg: str
    status: Status
    stdout: str
    stderr: str
    exit_code: int
```

#### 3.2.2 安全中间件

```python
# API Key配置
API_KEYS = {"your-secret-key-here"}

# IP限流配置
RATE_LIMIT = 60  # 每分钟请求数
ip_requests = {}

@app.middleware("http")
async def security_middleware(request: Request, call_next):
    # 1. API Key鉴权
    api_key = request.headers.get("X-API-Key")
    if not api_key or api_key not in API_KEYS:
        raise HTTPException(status_code=401, detail="未授权访问")
    
    # 2. IP频率限制
    client_ip = request.client.host
    current_time = time.time()
    
    if client_ip not in ip_requests:
        ip_requests[client_ip] = []
    
    # 清理60秒前的请求记录
    ip_requests[client_ip] = [
        t for t in ip_requests[client_ip] 
        if current_time - t < 60
    ]
    
    if len(ip_requests[client_ip]) >= RATE_LIMIT:
        raise HTTPException(status_code=429, detail="请求频率超限")
    
    ip_requests[client_ip].append(current_time)
    
    response = await call_next(request)
    return response
```

#### 3.2.3 执行接口实现

```python
@app.post("/execute", response_model=ExecuteResponse)
async def execute(request: ExecuteRequest):
    try:
        # 解析参数（兼容Judge0格式）
        input_params = request.stdin or ""
        params = input_params.split()
        
        # 构建C++程序调用命令
        cmd = ["./cpp_app"] + params
        
        # 执行C++程序
        result = subprocess.run(
            cmd,
            capture_output=True,
            text=True,
            timeout=10  # 超时限制
        )
        
        # 构建响应
        if result.returncode == 0:
            return ExecuteResponse(
                code=200,
                msg="调用成功",
                status=Status(id=1, description="Completed"),
                stdout=result.stdout.strip(),
                stderr="",
                exit_code=0
            )
        else:
            return ExecuteResponse(
                code=500,
                msg="C++程序执行失败",
                status=Status(id=2, description="Runtime Error"),
                stdout="",
                stderr=result.stderr.strip(),
                exit_code=result.returncode
            )
            
    except subprocess.TimeoutExpired:
        return ExecuteResponse(
            code=500,
            msg="程序执行超时",
            status=Status(id=4, description="Time Limit Exceeded"),
            stdout="",
            stderr="程序执行超过10秒",
            exit_code=-1
        )
    except Exception as e:
        return ExecuteResponse(
            code=400,
            msg="接口调用失败",
            status=Status(id=3, description="Bad Request"),
            stdout="",
            stderr=str(e),
            exit_code=-1
        )
```

### 3.3 Docker打包设计

#### 3.3.1 Dockerfile

```dockerfile
# 基础镜像
FROM alpine:3.19

# 安装依赖
RUN apk add --no-cache \
    g++ \
    python3 \
    py3-pip \
    gcc \
    make

# 设置工作目录
WORKDIR /app

# 复制C++源代码
COPY main.cpp .

# 编译C++程序
RUN g++ -std=c++11 -o cpp_app main.cpp

# 复制Python代码
COPY main.py .

# 安装Python依赖
RUN pip3 install --no-cache-dir fastapi>=0.100.0 uvicorn>=0.23.2

# 暴露端口
EXPOSE 4002

# 启动命令
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "4002"]
```

#### 3.3.2 构建流程

```bash
# 构建镜像
docker build -t cpp-api:v1 .

# 启动容器
docker run -d -p 4002:4002 --name cpp-api-server cpp-api:v1

# 查看日志
docker logs -f cpp-api-server
```

## 四、接口调用规范

### 4.1 基础信息

- **接口地址**：`http://[IP]:4002/execute`
- **请求方式**：POST
- **Content-Type**：application/json
- **认证方式**：请求头 `X-API-Key`

### 4.2 请求参数

```json
{
    "stdin": "1.5 2.5"  // C++程序输入参数，空格分隔
}
```

### 4.3 响应示例

#### 成功响应
```json
{
    "code": 200,
    "msg": "调用成功",
    "status": {
        "id": 1,
        "description": "Completed"
    },
    "stdout": "4",
    "stderr": "",
    "exit_code": 0
}
```

#### 错误响应
```json
{
    "code": 500,
    "msg": "C++程序执行失败",
    "status": {
        "id": 2,
        "description": "Runtime Error"
    },
    "stdout": "",
    "stderr": "参数错误：需要2个参数",
    "exit_code": 1
}
```

### 4.4 调用示例

#### curl命令
```bash
curl -X POST http://localhost:4002/execute \
  -H "X-API-Key: your-secret-key-here" \
  -H "Content-Type: application/json" \
  -d '{"stdin": "1.5 2.5"}'
```

#### Python代码
```python
import requests

url = "http://localhost:4002/execute"
headers = {
    "X-API-Key": "your-secret-key-here",
    "Content-Type": "application/json"
}
data = {"stdin": "1.5 2.5"}

response = requests.post(url, headers=headers, json=data)
print(response.json())
```

## 五、部署方案

### 5.1 本地开发部署

```bash
# 1. 编译C++程序
g++ -std=c++11 -o cpp_app main.cpp

# 2. 安装Python依赖
pip install fastapi uvicorn

# 3. 启动服务
uvicorn main:app --host 0.0.0.0 --port 4002
```

### 5.2 Docker部署

```bash
# 1. 构建镜像
docker build -t cpp-api:v1 .

# 2. 启动容器
docker run -d -p 4002:4002 --name cpp-api-server cpp-api:v1

# 3. 验证服务
docker ps
# 输出应包含 cpp-api-server 容器，状态为 Up
```

### 5.3 服务器部署注意事项

1. **端口配置**：确保服务器安全组放行4002端口
2. **Docker版本**：要求≥20.10.0
3. **资源配置**：建议1核2GB以上内存
4. **安全加固**：
   - 定期更新API Key
   - 限制访问来源IP
   - 禁止容器以root用户运行

## 七、故障排查指南

### 7.1 常见问题

| 问题现象 | 可能原因 | 解决方案 |
|---------|---------|---------|
| 容器启动失败 | 端口被占用 | 修改端口映射或释放4002端口 |
| 接口返回401 | API Key错误 | 检查请求头中的X-API-Key |
| 接口返回429 | 请求频率超限 | 降低请求频率或调整限流阈值 |
| C++程序执行失败 | 参数格式错误 | 检查输入参数是否符合要求 |
| 响应时间过长 | C++程序耗时 | 优化C++业务逻辑或增加超时时间 |
